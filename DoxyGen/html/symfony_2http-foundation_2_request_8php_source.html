746  public static function normalizeQueryString($qs)
747  {
748  if ('' == $qs) {
749  return '';
750  }
751 
752  $parts = [];
753  $order = [];
754 
755  foreach (explode('&', $qs) as $param) {
756  if ('' === $param || '=' === $param[0]) {
757  // Ignore useless delimiters, e.g. "x=y&".
758  // Also ignore pairs with empty key, even if there was a value, e.g. "=value", as such nameless values cannot be retrieved anyway.
759  // PHP also does not include them when building _GET.
760  continue;

767  // RFC 3986 with rawurlencode.
768  $parts[] = isset($keyValuePair[1]) ?
769  rawurlencode(urldecode($keyValuePair[0])).'='.rawurlencode(urldecode($keyValuePair[1])) :
